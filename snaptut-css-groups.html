Snap transforms on grouped groups

Lets fiddle with Snap matrices on groups.
Useful Matrix links w3.org svg coords and matrix


var s = Snap("#svgout"); 

var myOpacityStyle = 'opacity: 0.8;';
var myTransformStyle = '-webkit-transform: rotate(45deg);-moz-transform: rotate(45deg);-o-transform: rotate(45deg);-ms-transform: rotate(45deg);transform: rotate(45deg);';
var myStyle = myOpacityStyle;


var r0 = s.rect(150,150,100,100,20,20).attr({ fill: "orange", stroke: "black", strokeWidth: "2", style: myTransformStyle});
var c0 = s.circle(225,225,10).attr({ fill: "silver", stroke: "black" });
var g0 = s.group( r0,c0 );
//g0.animate({ 'transform' : 'r360,225,225' },4000);
g0.drag();
