Checking a handler with multiple elements




var s = Snap("#svgout"), c, cx, xy;

for( var count = 0; count < 500; count++ ) {
	cx = Math.random() * 800; cy = Math.random() * 400;

	c = s.circle( cx,cy,5 ).attr({ fill: "silver", id: 'id-' + count });

}

s.hover( function( ev ) { 
//	Couple of options of how to get access to the element
//	this.select( "#" + ev.target.id ).attr({ fill: 'red' }); OR
	var myElement = Snap( ev.target );
	myElement.attr({ fill: 'red' });
	s.text(myElement.attr('cx'), myElement.attr('cy') - 5, myElement.attr('id'));
	console.log( this, ev );
} );
