Using a Mask in Snap

Draw a big circle as a mask and animate objects


var s = Snap("#svgout"); 
//var rect = s.rect(0,0,400,400).attr({ fill: "blue" });
var bigC = s.circle(100,100,75).attr({ stroke: 'white', 'strokeWidth': 40, fill: 'black' });
var bigC2 = s.circle(250,250,75).attr({ stroke: 'silver', 'strokeWidth': 40, fill: 'black' });
var clipG = s.group(bigC,bigC2);

var r = s.rect(100,100,100,100,20,20).attr({ stroke: '#123456', 'strokeWidth': 20, fill: 'white' });
var c = s.circle(50,50,50).attr({ stroke: '#123456', 'strokeWidth': 20, fill: 'black' });

var g = s.group(r,c); 

g.attr({ mask: clipG });

g.animate({ transform: 'r360,150,150' }, 3000, mina.bounce );
clipG.animate({ transform: 't200,0' }, 3000, mina.bounce, function() { clipG.animate({ transform: 't0,0' }, 3000, mina.bounce)  }  );
