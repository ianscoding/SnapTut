Snap Hover with custom handlers

Just testing a closer on a hover


var s = Snap("#svgout");

for (var k = 1; k<4; k++) {

    var iCircleID = "iCircle_" + "test" + "_" + "whatever_j_is" + "_" + k;
    var iCircle = s.circle(k*50+50, k*50+50, 20).attr({
        "id": iCircleID
    });
    var iMark = s.text(k*50 + 50, k*50 + 50, "i");

    var infoGroupID = "infoGroup_" + "test" + "_" + "whatever_j_is" + "_" + k;
    var infoGroup = s.g(iCircle, iMark).attr( {
        "id": infoGroupID
    });
   
    (function() { 
       var enclosed_id = "#" + iCircleID;
       Snap.select("#" + infoGroupID).hover(
            function() { // f_in handler
                Snap.select(enclosed_id).attr({"fill": "#fcf0c3"});
            },
            function() { // f_out handler
                Snap.select(enclosed_id).attr({"fill": "#f4d887"});
            }
    ).click(
            function () {
                console.log('clicking' + enclosed_id);
            }
     );
   })();
}
