Using an image as a mask

Lets get an image and use it as a mask.


var s = Snap("#svgout"); 

var myCircle = s.circle(100,100,100).attr({ fill: 'red' });;

setInterval( function() {
		var myBitmap = s.image('test.bmp', 50,50, 200,200);
		myCircle.attr({ mask: myBitmap });	
			}, 2000 );
