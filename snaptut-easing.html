Click Handler starting animation

Here's an example click handler starting an animation


var s = Snap("#svgout"); 

var r = s.rect(100,100,100,100,20,20).attr({ stroke: '#123456', 'strokeWidth': 20, fill: 'red' });
var c = s.circle(50,50,50).attr({ stroke: '#123456', 'strokeWidth': 20, fill: 'blue' });

var g = s.group(r,c);

var easeOutCubic = function (t, b, c, d) {
	t /= d;
	t--;
	return c*(t*t*t + 1) + b;
};

var cubic = function (time, beg, diff, dur) {
                return diff*((time=time/dur-1)*time*time + 1) + beg;
}

g.animate({ transform: 'r45,150,150' }, 1000, "cubic-bezier(0,1,1,0)" );
